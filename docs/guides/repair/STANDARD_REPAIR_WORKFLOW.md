## 二、修复执行流程

### 2.1 修复脚本（标准模板）

```python
#!/usr/bin/env python3
"""
标准修复脚本 - ernchern/personal_info_classification 案例
任务类型: text-classification
模型类型: BERT
"""

import os
import sys
import json
import torch

# ===== 环境配置 =====
os.environ["HF_ENDPOINT"] = "https://hf-mirror.com"
os.environ["CUDA_VISIBLE_DEVICES"] = ""
os.environ["GRAPH_NET_EXTRACT_WORKSPACE"] = "/work/graphnet_workspace"

sys.path.insert(0, "/root/GraphNet")
from graph_net.torch.extractor import extract
from transformers import AutoConfig, AutoModelForSequenceClassification

# ===== 模型配置 =====
MODEL_ID = "ernchern/personal_info_classification"
MODEL_NAME_SAFE = "ernchern_personal_info_classification"
TASK_TYPE = "text-classification"
MODEL_DIR = f"/work/graphnet_workspace/huggingface/{TASK_TYPE}/{MODEL_NAME_SAFE}"

def repair_model():
    """标准修复流程"""
    print(f"=== 开始修复: {MODEL_ID} ===")

    # Step 1: 加载配置（只下载config.json，不下载权重）
    print("[1/6] 加载模型配置...")
    config = AutoConfig.from_pretrained(MODEL_ID, trust_remote_code=True)
    model_type = getattr(config, 'model_type', 'unknown')
    print(f"  模型类型: {model_type}")
    print(f"  隐藏层: {getattr(config, 'num_hidden_layers', 'N/A')}")
    print(f"  词表大小: {getattr(config, 'vocab_size', 'N/A')}")

    # Step 2: 随机初始化模型（祖训：不下载HF权重！）
    print("[2/6] 随机初始化模型...")
    model = AutoModelForSequenceClassification.from_config(
        config,
        trust_remote_code=True
    )
    model.eval()

    # Step 3: 准备输入数据
    print("[3/6] 准备输入数据...")
    vocab_size = getattr(config, 'vocab_size', 32000)
    seq_len = min(getattr(config, 'max_position_embeddings', 512), 128)  # 限制长度

    inputs = {
        "input_ids": torch.randint(0, vocab_size, (1, seq_len), dtype=torch.int64),
        "attention_mask": torch.ones((1, seq_len), dtype=torch.int64),
    }
    print(f"  输入shape: {inputs['input_ids'].shape}")

    # Step 4: 使用GraphNet抽取计算图
    print("[4/6] 抽取计算图...")

    # 设置临时工作目录
    temp_workspace = f"/work/graphnet_workspace/temp_{MODEL_NAME_SAFE}"
    os.makedirs(temp_workspace, exist_ok=True)
    os.environ["GRAPH_NET_EXTRACT_WORKSPACE"] = temp_workspace

    # 使用extract wrapper
    wrapped = extract(name=MODEL_NAME_SAFE, dynamic=True)(model)

    with torch.no_grad():
        outputs = wrapped(**inputs)

    print("  ✓ 抽取完成")

    # Step 5: 验证生成的文件
    print("[5/6] 验证生成文件...")
    temp_model_dir = os.path.join(temp_workspace, MODEL_NAME_SAFE)

    required_files = {
        "model.py": False,
        "weight_meta.py": False,
        "input_meta.py": False,
        "graph_net.json": False,
        "graph_hash.txt": False,
    }

    # 检查根目录和subgraph_0
    for fname in required_files:
        if os.path.exists(os.path.join(temp_model_dir, fname)):
            required_files[fname] = True
        elif os.path.exists(os.path.join(temp_model_dir, "subgraph_0", fname)):
            required_files[fname] = True

    for fname, exists in required_files.items():
        status = "✓" if exists else "✗"
        print(f"  {status} {fname}")

    # Step 6: 移动文件到正确位置
    print("[6/6] 移动文件到目标目录...")

    # 创建目标目录
    os.makedirs(MODEL_DIR, exist_ok=True)

    # 移动文件
    import shutil
    for item in os.listdir(temp_model_dir):
        src = os.path.join(temp_model_dir, item)
        dst = os.path.join(MODEL_DIR, item)
        if os.path.exists(dst):
            if os.path.isdir(dst):
                shutil.rmtree(dst)
            else:
                os.remove(dst)
        shutil.move(src, dst)
        print(f"  移动: {item}")

    # 生成 extract.py（祖训要求）
    extract_py_content = f'''#!/usr/bin/env python3
"""
Extract script for {MODEL_ID}
Task: {TASK_TYPE}
Generated by GraphNet Repair Agent
"""

import os
import sys
import torch

os.environ["HF_ENDPOINT"] = "https://hf-mirror.com"
os.environ["GRAPH_NET_EXTRACT_WORKSPACE"] = "/work/graphnet_workspace"

sys.path.insert(0, "/root/GraphNet")
from graph_net.torch.extractor import extract
from transformers import AutoConfig, AutoModelForSequenceClassification

MODEL_ID = "{MODEL_ID}"

def main():
    # 从config加载（祖训：不下载权重）
    config = AutoConfig.from_pretrained(MODEL_ID, trust_remote_code=True)
    model = AutoModelForSequenceClassification.from_config(config, trust_remote_code=True)
    model.eval()

    # 准备输入
    vocab_size = getattr(config, 'vocab_size', 32000)
    seq_len = min(getattr(config, 'max_position_embeddings', 512), 128)
    inputs = {{
        "input_ids": torch.randint(0, vocab_size, (1, seq_len), dtype=torch.int64),
        "attention_mask": torch.ones((1, seq_len), dtype=torch.int64),
    }}

    # 抽取
    wrapped = extract(name="{MODEL_NAME_SAFE}", dynamic=True)(model)
    with torch.no_grad():
        wrapped(**inputs)

    print("Extraction completed!")

if __name__ == "__main__":
    main()
'''
    extract_py_path = os.path.join(MODEL_DIR, "extract.py")
    with open(extract_py_path, "w") as f:
        f.write(extract_py_content)
    os.chmod(extract_py_path, 0o755)
    print(f"  生成: extract.py")

    # 清理临时目录
    shutil.rmtree(temp_workspace)

    print("\n=== 修复完成 ===")
    return True

if __name__ == "__main__":
    repair_model()
```

## 五、常见问题与解决方案

### 5.1 model.py 为空的解决方案

**现象**：
```
-rw-r--r-- 1 root root 0 Mar 29 21:08 model.py
```

**原因**：
- 未正确执行 forward pass
- extract wrapper 未正确触发
- 工作目录设置错误

**解决**：
1. 确认 `GRAPH_NET_EXTRACT_WORKSPACE` 环境变量设置正确
2. 确认执行了 `wrapped(**inputs)` 且有输出
3. 检查 extract 函数的返回值

### 5.2 模型加载失败

**现象**：
```
OSError: ernchern/personal_info_classification does not appear to have a file named config.json
```

**解决**：
```python
# 使用本地目录加载
config_path = MODEL_DIR  # 本地目录
config = AutoConfig.from_pretrained(config_path, trust_remote_code=True)
```

### 5.3 内存不足 (OOM)

**解决**：
```python
# 1. 使用CPU
os.environ["CUDA_VISIBLE_DEVICES"] = ""

# 2. 减小序列长度
seq_len = 32  # 而不是512

# 3. 批量大小设为1
inputs = {"input_ids": torch.randint(0, vocab_size, (1, seq_len))}
```

## 六、修复完成标准

### 6.1 文件完整性清单

| 文件 | 必需 | 说明 |
|------|------|------|
| `graph_hash.txt` | ✅ | 计算图哈希（16位hex字符串） |
| `model.py` | ✅ | 完整计算图定义（>1KB，包含GraphModule） |
| `weight_meta.py` | ✅ | 权重元数据（包含tensor shape/dtype） |
| `input_meta.py` | ✅ | 输入元信息 |
| `graph_net.json` | ✅ | 计算图元数据 |
| `extract.py` | ✅ | 复现脚本（可执行，包含完整流程） |
| `config.json` | ⚠️ | 模型配置。**非必须**：只有需要从HuggingFace重新加载模型时才需要；timm等模型可直接从结构推断配置 |

**config.json 说明**：
- HuggingFace 模型：需要 config.json 才能使用 `AutoConfig.from_pretrained()`
- timm/torchvision 模型：可直接从模型结构推断，不需要 config.json
- 如果缺失 config.json 但有完整的 model.py/weight_meta.py，仍视为完整模型

### 6.2 内容质量检查

- [ ] model.py 包含 `class GraphModule` 定义
- [ ] model.py 包含 `fx::` 或 `torch.fx` 相关节点
- [ ] weight_meta.py 包含详细的 `torch.Size` 信息
- [ ] extract.py 可以独立运行并复现结果

## 七、批量修复脚本模板

```python
#!/usr/bin/env python3
"""
批量修复脚本 - 基于标准流程
"""

import os
import sys

# 读取模型列表
MODEL_LIST_FILE = "/tmp/my_fillmask.txt"
TASK_TYPE = "fill-mask"

def read_model_list(filepath, start=0, end=70):
    """读取模型列表的指定范围"""
    with open(filepath) as f:
        models = [line.strip() for line in f if line.strip()]
    return models[start:end]

def repair_batch():
    """批量修复"""
    models = read_model_list(MODEL_LIST_FILE, 0, 70)

    success_count = 0
    fail_count = 0

    for model_name in models:
        print(f"\n{'='*60}")
        print(f"修复: {model_name}")
        print('='*60)

        try:
            # 调用标准修复流程
            # ... 修复代码 ...
            success_count += 1
        except Exception as e:
            print(f"失败: {e}")
            fail_count += 1

    print(f"\n=== 统计 ===")
    print(f"成功: {success_count}")
    print(f"失败: {fail_count}")

if __name__ == "__main__":
    repair_batch()
```
